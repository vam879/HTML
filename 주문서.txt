    <script>
        // 체크박스 선택 시 수량 입력창 활성화 제어
        const products = [
            { checkboxId: 'guatemala', qtyId: 'qty_guatemala' },
            { checkboxId: 'mandeling', qtyId: 'qty_mandeling' },
            { checkboxId: 'vinacafe', qtyId: 'qty_vinacafe' },
        ];

        products.forEach(p => {
            const checkbox = document.getElementById(p.checkboxId);
            const qtyInput = document.getElementById(p.qtyId);

            checkbox.addEventListener('change', () => {
                qtyInput.disabled = !checkbox.checked;
                if (!checkbox.checked) {
                    qtyInput.value = 0;
                } else if (qtyInput.value == 0) {
                    qtyInput.value = 1;
                }
            });
        });
